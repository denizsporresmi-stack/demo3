## 2. OYUN ENTEGRASYON TEKNİĞİ

### 2.1 iframe Yöntemi (En Hızlı)
```html
<!-- Mevcut sayfanın içine oyun embed etme -->
<iframe 
    id="game-frame"
    src="/games/flappy-bird/index.html"
    width="100%"
    height="600"
    scrolling="no"
    style="border: none; border-radius: 10px;">
</iframe>
```

### 2.2 postMessage API (Güvenli İletişim)
```javascript
// Ana sayfa → Oyun
document.getElementById('game-frame').contentWindow.postMessage({
    action: 'startGame',
    difficulty: 'hard'
}, '*');

// Oyun → Ana sayfa
window.parent.postMessage({
    action: 'gameFinished',
    score: 1250,
    timestamp: Date.now()
}, '*');

// Ana sayfa: İletiyi dinle
window.addEventListener('message', (event) => {
    if (event.data.action === 'gameFinished') {
        // Score'u veritabanına kaydet
        saveGameScore(event.data.score);
    }
});
```

### 2.3 Web Worker ile Paralel İşleme
```javascript
// heavy.js - Oyun mantığı
self.onmessage = function(e) {
    const gameLogic = e.data;
    const result = processGameLogic(gameLogic);
    self.postMessage(result);
};

// main.js
const worker = new Worker('heavy.js');
worker.postMessage(gameData);
worker.onmessage = (result) => {
    updateGameState(result.data);
};
```

### 2.4 API Entegrasyonu (Önerilir)
```javascript
// Oyun ID'si ile API çağrısı
async function loadGame(gameId) {
    const response = await fetch(`/api/games/${gameId}`);
    const game = await response.json();
    
    return {
        id: game.id,
        title: game.title,
        url: game.gameUrl,
        description: game.description,
        category: game.category,
        rating: game.rating,
        players: game.activePlayerCount
    };
}

// Oyun bittiğinde skor kaydet
async function saveGameScore(gameId, score, userId) {
    const response = await fetch('/api/scores', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            gameId,
            score,
            userId,
            timestamp: Date.now()
        })
    });
    return response.json();
}
```

db.games.createIndex({ category: 1 });
db.scores.createIndex({ gameId: 1, score: -1 });
db.scores.createIndex({ userId: 1, timestamp: -1 });

// Pagination
const LIMIT = 20;
const page = 1;
const skip = (page - 1) * LIMIT;

db.games.find()
    .skip(skip)
    .limit(LIMIT)
    .sort({ plays: -1 });

// Aggregation Pipeline
db.scores.aggregate([
    { $match: { gameId: ObjectId("...") } },
    { $group: { 
        _id: "$gameId",
        totalPlays: { $sum: 1 },
        avgScore: { $avg: "$score" }
    }},
    { $sort: { totalPlays: -1 } }
]);

## 6. GÜVENLİK UYGULAMALARI

### 6.1 Skor Manipülasyonunu Önleme
```javascript
// Backend'de doğrulama (Önemli!)
app.post('/api/scores', async (req, res) => {
    const { gameId, score } = req.body;
    const userId = req.user.id;
    
    // Skor mantıklı mı kontrol et
    if (score < 0 || score > MAX_POSSIBLE_SCORE[gameId]) {
        return res.status(400).json({ error: 'Invalid score' });
    }
    
    // Çok hızlı gelen skorları reddet
    const lastScore = await Score.findOne({ userId, gameId })
        .sort({ timestamp: -1 });
    
    if (lastScore && 
        Date.now() - lastScore.timestamp < MIN_TIME_BETWEEN_SCORES) {
        return res.status(400).json({ error: 'Too fast' });
    }
    
    // Skor kaydet
    const score = await Score.create({
        userId,
        gameId,
        score,
        hash: generateHash(score, userId, Date.now())
    });
    
    res.json(score);
});
```
